Overview

Procedure

On VICTORIA,

cd /etc/sysconfig/network-scripts
cat >ifcfg-eth1 <<DONE
# Intel Corporation 82571EB Gigabit Ethernet Controller
DEVICE=eth1
HWADDR=00:1B:21:D9:A1:AF
ONBOOT=yes
BRIDGE=xenbr0
NM_CONTROLLED=no
DONE
cat >ifcfg-xenbr0 <<DONE
DEVICE=xenbr0
TYPE=Bridge
BOOTPROTO=static
BROADCAST=192.168.1.255
IPADDR=192.168.1.104
NETMASK=255.255.255.0
ONBOOT=yes
DELAY=0
NM_CONTROLLED=no
DONE
service network restart
